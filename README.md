# react-scaled-content

[![npm badge][npm-badge-png]][package-url]

## Installation

```
npm install react-scaled-content
```

## Usage

```jsx
import { Scaled } from 'react-scaled-content';

function Sample () {
  return (
    <div style={{ width: 400, height: 300 }}>
      <Scaled>
        <div style={{ width: 600, height: 600 }}>
          ...
        </div>
      </Scaled>
    </div>  
  );
}
```
